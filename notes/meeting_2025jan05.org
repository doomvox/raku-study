**** peculiarity bruce gray turned up last meeting:

#+BEGIN_SRC sh
raku -n3 '{
   my (\k, \d, \f, \e) = .split("|");
   say join "|", k, d, $_, e for f.split(",");
}' file.txt

# Cannot resolve caller split(Any:U, Str:D); Routine does not have any
# candidates.  Is only the proto defined?

#+END_SRC 

#+BEGIN_SRC sh
 raku --target=ast -ne '
    my (\k, \d, \f, \e) = .split("|");
    say join "|", k, d, $_, e for f.split(",");
 ' file.txt
#+END_SRC
